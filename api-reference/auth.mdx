# API Authentication

The exchange uses **JWT** or **OAuth 2.0** tokens to authenticate and authorize API access.

To interact with the API, you must first obtain an **Access Token** by authenticating with your username and password.  
Once you have the token, include it in the `Authorization` header of all requests.
```Authorization: <your_access_token> ```
<CodeGroup>
```javascript Node.js (gRPC)
// This example uses grpc-web: https://github.com/grpc/grpc-web#readme
const {LoginRequest, LoginResponse} = require('./basic_auth_pb.js');
const {BasicAuthClient} = require('./basic_auth_grpc_web_pb.js');

// replace proxy_url with the ip of your cluster
var proxy_url = 'http://1.2.3.4'; 

var auth_client = new BasicAuthClient(proxy_url, null, null);

var login = new LoginRequest();
login.setUsername('admin');
login.setPassword('pass');

auth_client.login(login, {}, (err, response)=>{
    console.log(err);
    token = response.getToken();
    console.log(token);
    });
    call.on('error', function(err) {
        console.log("request error");
        console.log(err);
    });
    });
```

```json Curl
curl -X POST "https://api.polymarketexchange.com//auth/v1beta1/login" -H  "accept: application/json" \
-H  "Content-Type: application/json" \
-d "{  \"password\": \"your-password\",  \"username\": \"your-username\"}"
```
